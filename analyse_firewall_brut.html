Chargement et parsing du log brut¶

In [2]:
rows = []
total = 0
skipped = 0

with open(INPUT_LOG, "r", errors="replace") as f:
    for line in f:
        total += 1
        if not PATTERN_DATE.search(line):
            continue
        try:
            src_ip   = re.search(r'SRC=([\d\.]+)',  line).group(1)
            dst_ip   = re.search(r'DST=([\d\.]+)',  line).group(1)
            proto    = re.search(r'PROTO=(\w+)',    line).group(1)
            dst_port = int(re.search(r'DPT=(\d+)', line).group(1))
            rule_id  = re.search(r'RULE=(\d+)',    line).group(1)
            action   = "PERMIT" if "PERMIT" in line else "DENY"
            date_match = re.search(r'^([A-Z][a-z]{2}\s+\d+\s+\d+:\d+:\d+)', line)
            date = date_match.group(1) if date_match else "NA"
            rows.append({"timestamp": date, "src_ip": src_ip, "dst_ip": dst_ip,
                         "protocole": proto, "dst_port": dst_port, "action": action, "rule_id": rule_id})
        except Exception:
            skipped += 1
        if total % 1_000_000 == 0:
            print(f"  Lignes lues : {total:,} | Extraites : {len(rows):,}")

print(f"✅ {len(rows):,} lignes extraites sur {total:,} ({skipped:,} ignorées)")

Nettoyage et déduction des protocoles¶

df = pd.DataFrame(rows)

df["timestamp"] = pd.to_datetime(
    "2025 " + df["timestamp"].str.strip(),
    format="%Y %b %d %H:%M:%S", errors="coerce"
)
# Janvier et Février → 2026
df.loc[df["timestamp"].dt.month.isin([1, 2]), "timestamp"] += pd.DateOffset(years=1)
df["dst_port"] = pd.to_numeric(df["dst_port"], errors="coerce")

def deduce_protocol(row):
    proto = str(row["protocole"]).strip().upper()
    if proto == "TCP":  return "TCP"
    if proto == "UDP":  return "UDP"
    if proto == "ICMP": return "ICMP"
    dport = row["dst_port"]
    if pd.notna(dport):
        if dport in TCP_PORTS: return "TCP"
        if dport in UDP_PORTS: return "UDP"
    return "AUTRE" if proto in ("", "N/A", "NA", "NONE", "0") else proto

df["proto_deduit"] = df.apply(deduce_protocol, axis=1)

print("Répartition des protocoles déduits :")
